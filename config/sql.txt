CREATE TABLE `pupaa`.`users` (`USER_ID` VARCHAR(16) NOT NULL DEFAULT replace(convert(uuid() using utf8mb4),'-',''), `FIRST_NAME` VARCHAR(64) NOT NULL , `LAST_NAME` VARCHAR(64) NOT NULL , `EMAIL_ADDRESS` VARCHAR(64) NOT NULL , `DEPARTMENT` VARCHAR(32) NOT NULL , `PASSWORD` VARCHAR(64) NOT NULL , `LOGIN_ATTEMPTS` INT(8) NOT NULL DEFAULT '0' , `ACCOUNT_STATUS` INT(8) NOT NULL DEFAULT '1' , PRIMARY KEY (`USER_ID`)) ENGINE = InnoDB;
